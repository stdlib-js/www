ones

Create a "generic" array filled with ones.

Usage

var ones = require( '@stdlib/array/base/ones' );

ones( len )

Returns a "generic" array filled with ones.

var out = ones( 3 );
// returns [ 1.0, 1.0, 1.0 ]

Examples

var gscal = require( '@stdlib/blas/base/gscal' ).ndarray;
var ones = require( '@stdlib/array/base/ones' );

// Create a ones array:
var arr = ones( 10 );

// Scale element pairs...
gscal( 2, 2.0, arr, 1, 0 );
gscal( 2, 3.0, arr, 1, 2 );
gscal( 2, 4.0, arr, 1, 4 );
gscal( 2, 5.0, arr, 1, 6 );
gscal( 2, 6.0, arr, 1, 8 );

console.log( arr );
// => [ 2.0, 2.0, 3.0, 3.0, 4.0, 4.0, 5.0, 5.0, 6.0, 6.0 ]
