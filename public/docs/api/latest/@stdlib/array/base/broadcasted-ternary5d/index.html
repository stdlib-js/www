bternary5d

Apply a ternary callback to elements in three broadcasted nested input arrays and assign results to elements in a five-dimensional nested output array.

Usage

var bternary5d = require( '@stdlib/array/base/broadcasted-ternary5d' );

bternary5d( arrays, shapes, fcn )

Applies a ternary callback to elements in three broadcasted nested input arrays and assigns results to elements in a five-dimensional nested output array.

var add = require( '@stdlib/math/base/ops/add3' );
var zeros5d = require( '@stdlib/array/base/zeros5d' );

var x = [ [ [ 1.0, 2.0 ] ] ];
var y = [ [ [ 3.0 ], [ 4.0 ] ] ];
var z = [ [ [ [ 4.0, 5.0 ] ], [ [ 4.0, 5.0 ] ] ] ];
var out = zeros5d( [ 1, 1, 2, 2 ] );

var shapes = [
    [ 1, 1, 2 ],
    [ 1, 2, 1 ],
    [ 1, 2, 1, 2 ],
    [ 1, 1, 2, 2, 2 ]
];

bternary5d( [ x, y, z, out ], shapes, add );
// out => [ [ [ [ [ 8.0, 10.0 ], [ 9.0, 11.0 ] ], [ [ 8.0, 10.0 ], [ 9.0, 11.0 ] ] ] ] ]

The function accepts the following arguments:

  • arrays: array-like object containing three input nested arrays and one output nested array.
  • shapes: array shapes.
  • fcn: ternary function to apply.

Notes

  • The input and output array shapes must be broadcast compatible.

Examples

var discreteUniform = require( '@stdlib/random/base/discrete-uniform' ).factory;
var filled5dBy = require( '@stdlib/array/base/filled5d-by' );
var zeros5d = require( '@stdlib/array/base/zeros5d' );
var add = require( '@stdlib/math/base/ops/add3' );
var bternary5d = require( '@stdlib/array/base/broadcasted-ternary5d' );

var shapes = [
    [ 1, 1, 1, 1, 3 ],
    [ 1, 1, 3, 1, 1 ],
    [ 1, 1, 1, 3, 1 ],
    [ 1, 1, 3, 3, 3 ]
];

var x = filled5dBy( shapes[ 0 ], discreteUniform( -100, 100 ) );
console.log( x );

var y = filled5dBy( shapes[ 1 ], discreteUniform( -100, 100 ) );
console.log( y );

var z = filled5dBy( shapes[ 2 ], discreteUniform( -100, 100 ) );
console.log( z );

var out = zeros5d( shapes[ 3 ] );
console.log( out );

bternary5d( [ x, y, z, out ], shapes, add );
console.log( out );
