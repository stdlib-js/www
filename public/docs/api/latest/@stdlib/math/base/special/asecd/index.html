asecd

Compute the arcsecant (in degrees) of a double-precision floating-point number.

Usage

var asecd = require( '@stdlib/math/base/special/asecd' );

asecd( x )

Computes the arcsecant (in degrees) of a double-precision floating-point number.

var sqrt = require( '@stdlib/math/base/special/sqrt' );
var v = asecd( 1 );
// returns 0.0

v = asecd( 2 * sqrt( 3 ) / 3 );
// returns ~30.0

v = asecd( sqrt( 2 ) );
// returns ~45.0

v = asecd( 2 );
// returns ~60.0

v = asecd( Infinity );
// returns 90.0

v = asecd( NaN );
// returns NaN

Examples

var linspace = require( '@stdlib/array/base/linspace' );
var asecd = require( '@stdlib/math/base/special/asecd' );

var x = linspace( -1.0, 1.0, 100 );

var i;
for ( i = 0; i < x.length; i++ ) {
    console.log( asecd( x[ i ] ) );
}

C APIs

Usage

#include "stdlib/math/base/special/asecd.h"

stdlib_base_asecd( x )

Computes the arcsecant (in degrees) of a double-precision floating-point number.

double out = stdlib_base_asecd( 1.0 );
// returns 0.0

out = stdlib_base_asecd( 2.0 );
// returns ~60.0

The function accepts the following arguments:

  • x: [in] double input value.
double stdlib_base_asecd( const double x );

Examples

#include "stdlib/math/base/special/asecd.h"
#include <stdio.h>

int main( void ) {
    const double x[] = { 1.0, 1.45, 1.89, 2.33, 2.78, 3.22, 3.66, 4.11, 4.55, 5.0 };
    
    double v;
    int i;
    for ( i = 0; i < 10; i++ ) {
        v = stdlib_base_asecd( x[ i ] );
        printf( "asecd(%lf) = %lf\n", x[ i ], v );
    }
}
