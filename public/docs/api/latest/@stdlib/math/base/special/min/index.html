min

Return the minimum value.

Usage

var min = require( '@stdlib/math/base/special/min' );

min( x, y )

Returns the minimum value.

var v = min( 4.2, 3.14 );
// returns 3.14

v = min( +0.0, -0.0 );
// returns -0.0

If any argument is NaN, the function returns NaN.

var v = min( 4.2, NaN );
// returns NaN

v = min( NaN, 3.14 );
// returns NaN

Examples

var minstd = require( '@stdlib/random/base/minstd-shuffle' );
var min = require( '@stdlib/math/base/special/min' );

var x;
var y;
var v;
var i;

for ( i = 0; i < 100; i++ ) {
    x = minstd();
    y = minstd();
    v = min( x, y );
    console.log( 'min(%d,%d) = %d', x, y, v );
}

C APIs

Usage

#include "stdlib/math/base/special/min.h"

stdlib_base_min( x, y )

Returns the minimum value.

double out = stdlib_base_min( 4.2, 3.14 );
// returns 3.14

out = stdlib_base_min( 0.0, -0.0 );
// returns -0.0

The function accepts the following arguments:

  • x: [in] double input value.
  • y: [in] double input value.
double stdlib_base_min( const double x, const double y );

Examples

#include "stdlib/math/base/special/min.h"
#include <stdlib.h>
#include <stdio.h>

int main( void ) {
    double x;
    double y;
    double v;
    int i;

    for ( i = 0; i < 100; i++ ) {
        x = ( ( (double)rand() / (double)RAND_MAX ) * 200.0 ) - 100.0;
        y = ( ( (double)rand() / (double)RAND_MAX ) * 200.0 ) - 100.0;
        v = stdlib_base_min( x, y );
        printf( "x: %lf, y: %lf, min(x, y): %lf\n", x, y, v );
    }
}
