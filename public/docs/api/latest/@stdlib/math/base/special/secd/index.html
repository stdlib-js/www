secd

Compute the secant of an angle measured in degrees.

Usage

var secd = require( '@stdlib/math/base/special/secd' );

secd( x )

Computes the secant of x (in degrees).

var v = secd( 30 );
// returns ~1.15

v = secd( 45 );
// returns ~1.41

v = secd( 60 );
// returns ~2.0

v = secd( 90 );
// returns 16331239353195370

v = secd( 0 );
// returns 1.0

v = secd( NaN );
// returns NaN

Examples

var linspace = require( '@stdlib/array/base/linspace' );
var secd = require( '@stdlib/math/base/special/secd' );

var x = linspace( 1.1, 5.1, 100 );

var i;
for ( i = 0; i < x.length; i++ ) {
    console.log( secd( x[ i ] ) );
}

C APIs

Usage

#include "stdlib/math/base/special/secd.h"

stdlib_base_secd( x )

Computes the secant of x (in degrees).

double out = stdlib_base_secd( 30.0 );
// returns ~1.15

out = stdlib_base_secd( 45.0 );
// returns ~1.41

The function accepts the following arguments:

  • x: [in] double input value.
double stdlib_base_secd( const double x );

Examples

#include "stdlib/math/base/special/secd.h"
#include <stdio.h>

int main( void ) {
    const double x[] = { 0.0, 30.0, 45.0, 60.0, 90.0 };

    double y;
    int i;
    for ( i = 0; i < 5; i++ ) {
        y = stdlib_base_secd( x[ i ] );
        printf( "secd(%lf) = %lf\n", x[ i ], y );
    }
}
