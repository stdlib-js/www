Standard Deviation

Laplace distribution standard deviation.

The standard deviation for a Laplace random variable with location parameter mu and scale parameter b > 0 is

Usage

var stdev = require( '@stdlib/stats/base/dists/laplace/stdev' );

stdev( mu, b )

Returns the standard deviation for a Laplace distribution with location parameter mu and scale parameter b.

var y = stdev( 2.0, 1.0 );
// returns ~1.414

y = stdev( 0.0, 1.0 );
// returns ~1.414

y = stdev( -1.0, 4.0 );
// returns ~5.657

If provided NaN as any argument, the function returns NaN.

var y = stdev( NaN, 1.0 );
// returns NaN

y = stdev( 0.0, NaN );
// returns NaN

If provided b <= 0, the function returns NaN.

var y = stdev( 0.0, 0.0 );
// returns NaN

y = stdev( 0.0, -1.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var stdev = require( '@stdlib/stats/base/dists/laplace/stdev' );

var mu;
var b;
var y;
var i;

for ( i = 0; i < 10; i++ ) {
    mu = ( randu()*10.0 ) - 5.0;
    b = randu() * 20.0;
    y = stdev( mu, b );
    console.log( 'µ: %d, b: %d, SD(X;µ,b): %d', mu.toFixed( 4 ), b.toFixed( 4 ), y.toFixed( 4 ) );
}

C APIs

Usage

#include "stdlib/stats/base/dists/laplace/stdev.h"

stdlib_base_dists_laplace_stdev( mu, b )

Evaluates the standard deviation for a Laplace distribution with location parameter mu and scale parameter b.

double out = stdlib_base_dists_laplace_stdev( 0.0, 1.0 );
// returns ~1.414

The function accepts the following arguments:

  • mu: [in] double location parameter.
  • b: [in] double scale parameter.
double stdlib_base_dists_laplace_stdev( const double mu, const double b );

Examples

#include "stdlib/stats/base/dists/laplace/stdev.h"
#include <stdlib.h>
#include <stdio.h>

static double random_uniform( const double min, const double max ) {
    double v = (double)rand() / ( (double)RAND_MAX + 1.0 );
    return min + ( v*(max-min) );
}

int main( void ) {
    double mu;
    double b;
    double y;
    int i;

    for ( i = 0; i < 25; i++ ) {
        mu = random_uniform( -5.0, 5.0 );
        b = random_uniform( 0.0, 20.0 );
        y = stdlib_base_dists_laplace_stdev( mu, b );
        printf( "µ: %lf, b: %lf, SD(X;µ,b): %lf\n", mu, b, y );
    }
}
