Export assignment whileAsync

  • whileAsync(predicate: Predicate, fcn: Fcn, done: DoneCallback, thisArg?: any): void
    • Defined in lib/node_modules/@stdlib/utils/async/while/docs/types/index.d.ts:122

    Invokes a function while a test condition is true.

    Parameters

    • predicate: Predicate

      function which indicates whether to continue invoking a function

    • fcn: Fcn

      function to invoke

    • done: DoneCallback

      callback to invoke upon completion

    • Optional thisArg: any

      execution context for the invoked function

    Returns void

    Example

    function predicate( i, clbk ) {
    clbk( null, i < 5 );
}

function fcn( i, next ) {
    setTimeout( onTimeout, i );
    function onTimeout() {
        console.log( 'beep: %d', i );
        next();
    }
}

function done( error ) {
    if ( error ) {
        throw error;
    }
}

whileAsync( predicate, fcn, done );
